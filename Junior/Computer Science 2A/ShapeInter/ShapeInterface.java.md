/**

 * The Shape interface provides a contract for geometric shapes.

 * Any class that implements this interface must provide implementations

 * for calculating the area and perimeter of the shape.

 */

public interface ShapeInterface {

  

    /**

     * Calculates the area of the shape.

     *

     * @return the area of the shape as a double.

     */

    double area();

  

    /**

     * Calculates the perimeter of the shape.

     *

     * @return the perimeter of the shape as a double.

     */

    double perimeter();

}
